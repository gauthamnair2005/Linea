Functions

Define and use functions in Linea

Function Declaration

Basic Syntax

fn function_name(param1 @ type1, param2 @ type2) -> return_type {
    // function body
    return value
}

Simple Function

fn greet(name @ str) -> str {
    return "Hello, " + name
}

var result @ str = greet("Alice")
display result

Parameters

Multiple Parameters

fn add(a @ int, b @ int) -> int {
    return a + b
}

var sum @ int = add(10, 20)

No Parameters

fn get_pi() -> float {
    return 3.14159
}

No Return Value

fn print_message(msg @ str) {
    display msg
}

Return Values

Explicit Return

fn square(x @ int) -> int {
    return x * x
}

Implicit Return (Last Expression)

fn double(x @ int) -> int {
    x * 2  // Implicit return
}

Recursion

Factorial Example

fn factorial(n @ int) -> int {
    if n <= 1 {
        return 1
    }
    return n * factorial(n - 1)
}

display factorial(5)  // Output: 120

Fibonacci

fn fib(n @ int) -> int {
    if n <= 1 { return n }
    return fib(n-1) + fib(n-2)
}

Scoping

Functions have their own scope. Variables are local to the function:

fn compute() -> int {
    var local @ int = 42
    return local
}

display compute()  // Works
// display local  // Error: local not in scope

Common Patterns

Finding Maximum of Two Numbers

fn max(a @ int, b @ int) -> int {
    if a > b { return a }
    return b
}

Array Sum

fn sum_array(arr @ [int]) -> int {
    var sum @ int = 0
    for i from 0~(3) {
        sum = sum + arr[i]
    }
    return sum
}

Best Practices

  • Use descriptive names: Function names should describe what they do
  • Keep functions small: Each function should do one thing
  • Add comments: Document complex logic
  • Handle edge cases: Check for empty arrays, null values, etc.
  • Use type annotations: Always specify parameter and return types
