# Train a Neural Network in 10 lines
import ml

var model = ml::Linear(784, 10)
var data = datasets::load_csv("mnist.csv")

for epoch from 0~10 {
    var pred = ml::forward(model, data.x)
    var loss = ml::cross_entropy(pred, data.y)
    var grad = ml::backward(pred, data.y)
    
    ml::step(model, grad, 0.01)
    display "Loss: " + loss
}
