`webserver` and `framework` Libraries

Build quick HTTP services and scaffold Django-like projects with route-driven development.

Web Server

import webserver

# Serve one request then return (great for tests/smoke checks)
webserver::serveText("127.0.0.1", 8080, "Hello from Linea", 1)

Framework

import framework

framework::newProject("myapp")
framework::addRoute("myapp", "/", "Welcome")
framework::addRoute("myapp", "/health", "{\"ok\":true}")
display framework::routes("myapp")
framework::runDevServer("myapp", "127.0.0.1", 8080, 1)
