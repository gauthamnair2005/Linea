`git` Standard Library

Simple function-first wrappers around common Git actions.

Read-Only APIs

import git

display git::isRepo(".")
display git::currentBranch(".")
display git::lastCommit(".")
display git::status(".")
display git::log(".", 5)
display git::diff(".")

Action APIs

git::add(".", ".")
git::commit(".", "update docs")
git::push(".", "origin", "main")
git::pull(".", "origin", "main")
git::checkout(".", "feature/new-branch")
