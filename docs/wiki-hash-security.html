`hash` and `security` Modules

Native hashing, token generation, constant-time comparison, and password safety utilities.

Hash API

import hash

display hash::sha256("linea")
display hash::sha512("linea")
display hash::md5("linea")
display hash::withSalt("sha256", "linea", "fixed-salt")
display hash::secureEquals("abc", "abc")

Security API

import security

var token = security::randomToken(24)
var entropy = security::randomBytes(16)
display token
display entropy

var pw = "Linea#2026Secure"
var score = security::passwordScore(pw)
var strong = security::isStrongPassword(pw)
var stored = security::passwordHash(pw)
var ok = security::passwordVerify(pw, stored)
display score
display strong
display ok
